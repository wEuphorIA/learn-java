二分查找 基础版int[] ={1,2,3,4}

```java
public static int binarySearch(int[] nums,int target){
    int left = 0;
    int right = nums.length -1;
    while(left <= right){
        int mid = (left+right) >>> 1;
        if(nums[mid] < target){
            //如果目标值大于中间值，右指针移到中间值的右边
            left = mid +1;
        }else if(target < nums[mid]){
            //如果目标值小于中间值，右指针移到中间值的左边
            right = mid -1;
        }else{
            return mid;
        }
    }
    return -1;
}
```

    int left = 0;
    int right = nums.length; //改动1 
    while(left < right){     //改动2
        int mid = (left+right) >>> 1;
        if(nums[mid] < target){
            //如果目标值大于中间值，右指针移到中间值的右边
            left = mid +1;
        }else if(target < nums[mid]){
            //如果目标值小于中间值，右指针移到中间值的左边
            right = mid; //改动3
        }else{
            return mid;
        }
    }
    return -1;
}
